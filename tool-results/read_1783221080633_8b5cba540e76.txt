     5→// MARK: - Sync Engine (Production — Latency Compensated)
     6→/// Synchronizes AVPlayer playback across all room participants with
     7→/// **network latency compensation**.
     8→///
     9→/// ┌─────────────────────────────────────────────────────────────────┐
    10→/// │                   THE LATENCY PROBLEM                            │
    11→/// │                                                                  │
    12→/// │  Host presses Play at media-time = 10.0s, wall-clock T0.        │
    13→/// │  Command travels to server (T0 + uplink)                         │
    14→/// │  Server broadcasts to all clients (T0 + uplink + downlink)       │
    15→/// │  Client receives command at T0 + RTT                              │
    16→/// │                                                                  │
    17→/// │  If client seeks to 10.0s and plays, it is already RTT seconds   │
    18→/// │  behind the host. Everyone sees different frames.                │
    19→///                                                                  │
    20→/// │  SOLUTION: compensate using synchronized server clock + RTT.     │
    21→/// └─────────────────────────────────────────────────────────────────┘
    22→///
    23→/// Compensation formula:
    24→///   elapsedSinceEvent = currentServerTime - eventServerTimestamp
    25→///   targetMediaTime = eventMediaTime + (isPlayingEvent ? elapsedSinceEvent : 0)
    26→///
    27→/// The client's view of "currentServerTime" is kept accurate by the
    28→/// WebSocketClient's ping/pong clock-sync (see synchronizedServerTime).
    29→
    30→@MainActor
    31→final class SyncEngine: NSObject, ObservableObject, @unchecked Sendable {

   172→            roomID: roomID,
   173→            senderID: userID,
   174→            mediaTime: currentTime,
   175→            timestamp: currentServerTime

   202→    func seek(to time: TimeInterval) {
   203→        guard isHost else { return }
   204→        let clamped = max(0, min(time, duration))
   205→
   206→        let handler: @Sendable (Bool) -> Void = { [weak self] _ in
   207→            guard let self else { return }
   208→            Task { @MainActor in
   209→                self.currentTime = clamped
   210→                self.broadcastSyncCommand(.seek, mediaTime: clamped)
   211→                self.seekCompletionHandler = nil
   212→            }
   213→        }
   214→        seekCompletionHandler = handler
   215→        player?.seek(to: CMTime(seconds: clamped, preferredTimescale: 600),
   216→                     completionHandler: handler)
   217→
   218→        // 🔧 FIX 2.3: Fallback timeout — if seek doesn't complete in 2s, broadcast anyway
   219→        Task { @MainActor [weak self] in
   220→            try? await Task.sleep(nanoseconds: 2_000_000_000)
   221→            guard let self, self.seekCompletionHandler != nil else { return }
   222→            Logger.sync.warn("Seek timeout — broadcasting anyway")
   223→            self.currentTime = clamped
   224→            self.broadcastSyncCommand(.seek, mediaTime: clamped)
   225→            self.seekCompletionHandler = nil
   226→        }
   227→    }

   243→    // MARK: - Incoming Sync Command Handling (LATENCY COMPENSATED)
   244→
   245→    func handleSyncMessage(_ message: SyncMessage) {
   246→        // Ignore echoes of our own commands (server broadcasts to all incl. sender)
   247→        guard message.senderID != userID else { return }
   248→
   249→        switch message.command {
   250→        case .play:
   251→            handlePlay(message)
   252→        case .pause:
   253→            handlePause(message)
   254→        case .seek:
   255→            handleSeek(message)
   256→        case .changeMedia:
   257→            if let item = message.mediaItem {
   258→                handleMediaChange(item, from: message)
   259→            }
   260→        case .stateRequest:
   261→            // A participant is asking the host for the current state
   262→            if isHost {
   263→                respondWithCurrentState()
   264→            }
   265→        case .stateResponse:
   266→            handleStateResponse(message)
   267→        case .correction:
   268→            handleForcedCorrection(message)
   269→        case .ping, .pong:
   270→            break
   271→        }
   272→    }

   333→    /// Compare local playback position against the extrapolated host position.
   334→    /// If drift exceeds threshold, self-correct (participant) or request state (host).
   335→    private func checkDrift() {
   336→        guard lastSyncEventTime > 0 else { return }
   337→
   338→        // Extrapolate where the host SHOULD be right now
   339→        let elapsed = currentServerTime - lastSyncEventTime
   340→        let extrapolatedHostTime: TimeInterval
   341→        if lastSyncWasPlaying {
   342→            extrapolatedHostTime = min(lastSyncMediaTime + elapsed, duration)
   343→        } else {
   344→            extrapolatedHostTime = lastSyncMediaTime
   345→        }
   346→
   347→        let drift = abs(currentTime - extrapolatedHostTime)
   348→        let driftMs = drift * 1000
   349→        syncQuality = .fromDrift(driftMs)
   350→
   351→        // Surface telemetry
   352→        estimatedRTTms = Int(estimatedRTT * 1000)
   353→
   354→        // Hard resync: participant drifted way off — request fresh state from host
   355→        if !isHost && drift > Constants.hardResyncThreshold {
   356→            Logger.sync.warn("Hard drift: \(String(format: "%.0f", driftMs))ms — requesting state from host")
   357→            requestStateFromHost()
   358→            return
   359→        }
   360→
   361→        // Soft correction: small drift, nudge locally without a visible jump
   362→        if !isHost && drift > Constants.driftThreshold && drift <= Constants.hardResyncThreshold {
   363→            Logger.sync.info("Soft drift: \(String(format: "%.0f", driftMs))ms — self-correcting to \(String(format: "%.2f", extrapolatedHostTime))s")
   364→            seekSilently(to: extrapolatedHostTime, preserveRate: isPlaying)
   365→            lastCompensationMs = Int(driftMs)
   366→        }
   367→    }

   369→    // MARK: - Latency-Compensated Command Handlers
   370→
   371→    /// PLAY command received.
   372→    /// The host stamped this with (mediaTime, serverTimestamp).
   373→    /// By the time we receive it, real playback on the host has advanced by
   374→    /// (currentServerTime - serverTimestamp). We must seek AHEAD by that delta
   375→    /// so we're aligned with where the host actually is right now.
   376→    private func handlePlay(_ message: SyncMessage) {
   377→        let eventMediaTime = message.mediaTime ?? 0
   378→        let eventServerTime = message.timestamp
   379→
   380→        // ─── Core latency compensation ───
   381→        let elapsedSinceEvent = max(0, currentServerTime - eventServerTime)
   382→        let compensatedTarget = min(
   383→            eventMediaTime + min(elapsedSinceEvent, Constants.maxPredictiveJump),
   384→            duration > 0 ? duration : .infinity
   385→        )
   386→
   387→        let compensationMs = Int(elapsedSinceEvent * 1000)
   388→        lastCompensationMs = compensationMs
   389→        recordSyncPoint(mediaTime: compensatedTarget, isPlaying: true, serverTime: currentServerTime)
   390→
   391→        Logger.sync.info("▶️ PLAY  host@\(fmt(eventMediaTime))s → seek to \(fmt(compensatedTarget))s (+\(compensationMs)ms latency, RTT \(Int(estimatedRTT*1000))ms)")
   392→
   393→        // 🔧 FAST PATH: If we're already playing and within tolerance, do nothing
   394→        // (avoids stutter on rapid play/pause toggles)
   395→        if isPlaying && abs(currentTime - compensatedTarget) < Constants.seekTolerance {
   396→            return
   397→        }
   398→
   399→        // 🔧 FAST PATH: If drift is small (< 2s), just play without seeking —
   400→        // the drift monitor will self-correct within a few seconds.
   401→        // This avoids the "seek then play" stutter that was causing delay.
   402→        if abs(currentTime - compensatedTarget) < 2.0 {
   403→            player?.play()
   404→            isPlaying = true
   405→            return
   406→        }
   407→
   408→        // Large drift — seek to compensated position, then play
   409→        player?.seek(to: CMTime(seconds: compensatedTarget, preferredTimescale: 600)) { [weak self] _ in
   410→            guard let self else { return }
   411→            Task { @MainActor in
   412→                self.player?.play()
   413→                self.isPlaying = true
   414→                self.currentTime = compensatedTarget
   415→            }
   416→        }
   417→    }

   419→    /// PAUSE command received.
   420→    /// 🔧 IMMEDIATE: Pause instantly for zero-latency visual sync.
   421→    /// Seek to exact frame happens after pause (non-blocking).
   422→    private func handlePause(_ message: SyncMessage) {
   423→        let eventMediaTime = message.mediaTime ?? currentTime
   424→        recordSyncPoint(mediaTime: eventMediaTime, isPlaying: false, serverTime: message.timestamp)
   425→
   426→        Logger.sync.info("⏸️ PAUSE at \(fmt(eventMediaTime))s")
   427→
   428→        // 🔧 IMMEDIATE pause — no waiting for seek completion
   429→        player?.pause()
   430→        isPlaying = false
   431→        currentTime = eventMediaTime
   432→
   433→        // Seek to exact paused frame (async, non-blocking)
   434→        player?.seek(to: CMTime(seconds: eventMediaTime, preferredTimescale: 600), toleranceBefore: .zero, toleranceAfter: .zero) { [weak self] _ in
   435→            Task { @MainActor in self?.currentTime = eventMediaTime }
   436→        }
   437→    }

   439→    /// SEEK command received (also used as periodic state pulse from host).
   440→    /// Compensate for latency so we land where the host actually is now.
   441→    ///
   442→    /// 🔧 FIX M1: Was using `elapsedSinceEvent < stateBroadcastInterval + 1` as a
   443→    /// heuristic to detect state pulses vs real seeks. But the host's periodic
   444→    /// state broadcast (every 2s) and a user-initiated seek can both arrive within
   445→    /// that 3s window, causing real seeks to be silently reverted (extrapolated
   446→    /// back to the original position).
   447→    ///
   448→    /// New heuristic: a state pulse is a seek where |eventMediaTime - lastSyncMediaTime|
   449→    /// is small (≤ 2s, normal playback progress). A real seek has a large jump.
   450→    /// This distinguishes "host is just broadcasting position" from "host actually
   451→    /// jumped to a new position".
   452→    private func handleSeek(_ message: SyncMessage) {
   453→        guard let eventMediaTime = message.mediaTime else { return }
   454→        let eventServerTime = message.timestamp
   455→        let elapsedSinceEvent = max(0, currentServerTime - eventServerTime)
   456→
   457→        // 🔧 FIX M1: Detect state pulse by comparing media time delta, NOT time elapsed.
   458→        // A state pulse has media time ≈ lastSyncMediaTime + (elapsed * playbackRate).
   459→        // A real seek has a large jump in media time relative to lastSyncMediaTime.
   460→        let expectedProgress = isPlaying ? elapsedSinceEvent : 0
   461→        let mediaTimeDelta = abs(eventMediaTime - (lastSyncMediaTime + expectedProgress))
   462→        let isStatePulse = mediaTimeDelta < Constants.stateBroadcastInterval  // ≤ 2s drift = pulse
   463→
   464→        let compensatedTarget: TimeInterval
   465→        if isStatePulse && isPlaying {
   466→            compensatedTarget = min(eventMediaTime + elapsedSinceEvent, duration > 0 ? duration : .infinity)
   467→        } else {
   468→            // Real seek — don't extrapolate, just go to the exact position.
   469→            compensatedTarget = eventMediaTime
   470→        }
   471→
   472→        // Within tolerance → ignore (prevents jitter from periodic pulses)
   473→        if abs(currentTime - compensatedTarget) < Constants.seekTolerance {
   474→            recordSyncPoint(mediaTime: compensatedTarget, isPlaying: isPlaying, serverTime: currentServerTime)
   475→            return
   476→        }
   477→
   478→        recordSyncPoint(mediaTime: compensatedTarget, isPlaying: isPlaying, serverTime: currentServerTime)
   479→        Logger.sync.info("⏩ SEEK  host@\(fmt(eventMediaTime))s → \(fmt(compensatedTarget))s [\(isStatePulse ? "pulse" : "real-seek")]")
   480→
   481→        let wasPlaying = isPlaying
   482→        seekSilently(to: compensatedTarget, preserveRate: wasPlaying)
   483→    }

   485→    private func handleMediaChange(_ item: MediaItem, from message: SyncMessage) {
   486→        Logger.sync.info("🎬 Media change: \(item.displayTitle)")
   487→        loadMedia(item)
   488→
   489→        // If host is already partway in, jump to their position
   490→        if let startTime = message.mediaTime, startTime > 0.1 {
   491→            player?.seek(to: CMTime(seconds: startTime, preferredTimescale: 600))
   492→            currentTime = startTime
   493→            recordSyncPoint(mediaTime: startTime, isPlaying: false, serverTime: message.timestamp)
   494→        }
   495→    }
   496→
   497→    private func handleStateResponse(_ message: SyncMessage) {
   498→        guard !isHost, let hostMediaTime = message.mediaTime else { return }
   499→
   500→        // Host responded to our stateRequest with its current position + server time.
   501→        // Apply full latency compensation.
   502→        let elapsedSinceEvent = max(0, currentServerTime - message.timestamp)
   503→        let hostPlaying = message.command == .play  // host marks play vs seek
   504→        let target = hostPlaying
   505→            ? min(hostMediaTime + elapsedSinceEvent, duration > 0 ? duration : .infinity)
   506→            : hostMediaTime
   507→
   508→        let drift = abs(currentTime - target)
   509→        if drift > Constants.driftThreshold {
   510→            Logger.sync.warn("State response: correcting \(String(format: "%.0f", drift*1000))ms → \(fmt(target))s")
   511→            seekSilently(to: target, preserveRate: hostPlaying)
   512→            recordSyncPoint(mediaTime: target, isPlaying: hostPlaying, serverTime: currentServerTime)
   513→        }
   514→    }
   515→
   516→    private func handleForcedCorrection(_ message: SyncMessage) {
   517→        guard let target = message.mediaTime else { return }
   518→        Logger.sync.info("🔄 Forced correction → \(fmt(target))s")
   519→        seekSilently(to: target, preserveRate: isPlaying)
   520→        recordSyncPoint(mediaTime: target, isPlaying: isPlaying, serverTime: currentServerTime)
   521→        syncQuality = .perfect
   522→    }

   524→    // MARK: - Seek Helpers
   525→
   526→    /// Seek without a visible stutter — used for drift correction.
   527→    /// Uses tolerant seeking to avoid rebuffering on tiny adjustments.
   528→    private func seekSilently(to time: TimeInterval, preserveRate: Bool) {
   529→        let clamped = max(0, min(time, duration))
   530→        let tolerance = CMTime(seconds: Constants.seekTolerance, preferredTimescale: 600)
   531→        player?.seek(
   532→            to: CMTime(seconds: clamped, preferredTimescale: 600),
   533→            toleranceBefore: tolerance,
   534→            toleranceAfter: tolerance,
   535→            completionHandler: { [weak self] _ in
   536→                guard let self else { return }
   537→                Task { @MainActor in self.currentTime = clamped }
   538→            }
   539→        )
   540→        if preserveRate { player?.play() }
   541→    }

   575→    // MARK: - Broadcast Helpers
   576→
   577→    private func broadcastSyncCommand(_ command: SyncCommand, mediaTime: TimeInterval) {
   578→        let msg = SyncMessage(
   579→            command: command,
   580→            roomID: roomID,
   581→            senderID: userID,
   582→            mediaTime: mediaTime,
   583→            timestamp: currentServerTime
   584→        )
   585→        broadcast(msg)
   586→    }
   587→
   588→    private func broadcast(_ message: SyncMessage) {
   589→        // Inject synchronized server time into every outgoing command
   590→        var msg = message
   591→        if msg.timestamp == 0 { msg.timestamp = currentServerTime }
   592→
   593→        if let data = try? JSONEncoder().encode(msg),
   594→           let string = String(data: data, encoding: .utf8) {
   595→            wsClient.send(string)
   596→        }
   597→    }

   601→    private func addTimeObserver() {
   602→        guard let player else { return }
   603→        let interval = CMTime(seconds: 0.25, preferredTimescale: 600)
   604→        timeObserver = player.addPeriodicTimeObserver(forInterval: interval, queue: .main) { [weak self] time in
   605→            guard let self, time.seconds.isFinite else { return }
   606→            Task { @MainActor in self.currentTime = time.seconds }
   607→        }
   608→    }
   609→
   610→    private func teardownPlayer() {
   611→        if let observer = timeObserver {
   612→            player?.removeTimeObserver(observer)
   613→            timeObserver = nil
   614→        }
   615→        // 🔧 FIX 1.2: Remove buffer observer on teardown
   616→        if let observer = bufferObserver {
   617→            NotificationCenter.default.removeObserver(observer)
   618→            bufferObserver = nil
   619→        }
   620→        player?.pause()
   621→        player?.replaceCurrentItem(with: nil)
   622→        player = nil
   623→        playerItem = nil
   624→        currentMediaItem = nil
   625→        isPlaying = false
   626→        currentTime = 0
   627→        duration = 0
   628→        stopDriftMonitor()
   629→        stopStateBroadcast()
   630→    }
   631→
   632→    func cleanup() {
   633→        teardownPlayer()
   634→    }

   642→    private func observeStatus(_ item: AVPlayerItem) {
   643→        item.publisher(for: \.status, options: .new)
   644→            .sink { [weak self] status in
   645→                switch status {
   646→                case .readyToPlay:
   647→                    self?.isLoadingMedia = false
   648→                    self?.errorMessage = nil
   649→                case .failed:
   650→                    self?.errorMessage = item.error?.localizedDescription ?? "Failed to load media"
   651→                    self?.isLoadingMedia = false
   652→                    Logger.sync.error("AVPlayerItem failed: \(self?.errorMessage ?? "unknown")")
   653→                default:
   654→                    break
   655→                }
   656→            }
   657→            .store(in: &cancellables)
   658→    }

   660→    // 🔧 FIX 1.2: Buffer underrun observer — detects when AVPlayer stalls
   661→    // due to slow network. Pauses local playback (without broadcasting pause
   662→    // to other participants), then resumes + requests fresh state from host
   663→    // when buffer recovers. Prevents visible desync during buffering.
   664→    private func observeBufferUnderrun(_ item: AVPlayerItem) {
   665→        // Listen for playback stall notification
   666→        bufferObserver = NotificationCenter.default.addObserver(
   667→            forName: .AVPlayerItemPlaybackStalled,
   668→            object: item,
   669→            queue: .main
   670→        ) { [weak self] _ in
   671→            guard let self, !self.isHost else { return }
   672→            Task { @MainActor in
   673→                self.bufferUnderrunCount += 1
   674→                Logger.sync.warn("Buffer underrun #\(self.bufferUnderrunCount) — local pause (no broadcast)")
   675→                self.player?.pause()
   676→            }
   677→        }
   678→
   679→        // Also observe isPlaybackLikelyToKeepUp for recovery
   680→        item.publisher(for: \.isPlaybackLikelyToKeepUp, options: .new)
   681→            .receive(on: DispatchQueue.main)
   682→            .sink { [weak self] canKeepUp in
   683→                guard let self, !self.isHost else { return }
   684→                if canKeepUp && self.isPlaying {
   685→                    // Buffer recovered — resume local playback + request fresh state
   686→                    Logger.sync.info("Buffer recovered — resuming + requesting fresh state from host")
   687→                    self.player?.play()
   688→                    self.requestStateFromHost()
   689→                }
   690→            }
   691→            .store(in: &cancellables)
   692→    }
