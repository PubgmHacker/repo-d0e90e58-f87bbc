   579→    private let reactionThrottle: TimeInterval = 0.5

   581→    private func triggerReaction(emoji: String) {
   582→        let now = Date()
   583→        guard now.timeIntervalSince(lastReactionTime) >= reactionThrottle else { return }
   584→        lastReactionTime = now
   585→        HapticManager.impact(.soft)
   586→        reactionTrigger = ReactionTrigger(
   587→            point: CGPoint(x: UIScreen.main.bounds.width / 2,
   588→                          y: UIScreen.main.bounds.height * 0.35),
   589→            emoji: emoji
   590→        )
   591→        syncManager?.sendReaction(emoji: emoji, senderId: viewModel?.currentUserId ?? "unknown", senderName: viewModel?.room.hostName ?? "You")
   592→    }

   603→    private func sendMessage() {
   604→        guard let viewModel, let syncManager else { return }
   605→        let text = viewModel.chatText.trimmingCharacters(in: .whitespaces)

   665→        // This updates the seek bar + time display without triggering seeks.

   712→            if !vm.syncEngine.isPlaying { vm.syncEngine.togglePlayPause() }

   716→            if vm.syncEngine.isPlaying { vm.syncEngine.togglePlayPause() }
   717→        }
   718→        manager.onSeekCommand = { pos in vm.syncEngine.seek(to: pos) }
   719→        manager.onReactionReceived = { HapticManager.impact(.soft) }

   721→
   722→        viewModel = vm
   723→        syncManager = manager
   724→        self.voiceChat = voiceChat
