   545→    private func startCTACollapseTimer() {
   546→        ctaCollapseTimer?.invalidate()
   547→        ctaCollapseTimer = Timer.scheduledTimer(withTimeInterval: 6.0, repeats: false) { _ in
   548→            withAnimation(.spring(response: 0.6, dampingFraction: 0.7)) {
   549→                isCTACollapsed = true
   550→            }
   551→        }
   552→    }

   554→    private func resetCTACollapseTimer() {
   555→        ctaCollapseTimer?.invalidate()
   556→        if isCTACollapsed {
   557→            withAnimation(.spring(response: 0.6, dampingFraction: 0.7)) {
   558→                isCTACollapsed = false
   559→            }
   560→        }
   561→        startCTACollapseTimer()
   562→    }

   577→        // Сортируем по активности, берём разные сервисы
   578→        let all = viewModel.filteredRooms.sorted { $0.participantCount > $1.participantCount }
   579→        // Группируем по сервису чтобы получить разнообразие
   580→        var seen = Set<VideoService>()
   581→        var result: [Room] = []
   582→        for room in all {
   583→            if let source = room.mediaItem?.source, let service = VideoService(rawValue: source.rawValue), !seen.contains(service) {
   584→                seen.insert(service)
   585→                result.append(room)
   586→            }
   587→        }
   588→        // Добавляем остальные
   589→        result.append(contentsOf: all.filter { room in
   590→            !result.contains(where: { $0.id == room.id })
   591→        })
   592→        return Array(result.prefix(10))
   593→    }

   666→    private var isValidJoinInput: Bool {
   667→        let trimmed = joinInput.trimmingCharacters(in: .whitespaces)
   668→        return trimmed.count == 6 || trimmed.contains("http") || trimmed.contains("plnk") || trimmed.contains(".")
   669→    }
   670→
   671→    private func extractCode(from input: String) -> String {
   672→        let trimmed = input.trimmingCharacters(in: .whitespaces)
   673→        if trimmed.count == 6 && trimmed.allSatisfy({ $0.isLetter || $0.isNumber }) { return trimmed }
   674→        if let url = URL(string: trimmed) {
   675→            if let code = URLComponents(url: url, resolvingAgainstBaseURL: false)?.queryItems?.first(where: { $0.name == "code" })?.value { return code }
   676→            let path = url.lastPathComponent
   677→            if path.count >= 4 && path.count <= 8 { return path.uppercased() }
   678→        }
   679→        return trimmed
   680→    }
