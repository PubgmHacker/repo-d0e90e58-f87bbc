     5→// MARK: - App Language
     6→/// Поддерживаемые языки приложения. Переключаются в рантайме (не системно).
     7→enum AppLanguage: String, CaseIterable, Identifiable, Codable {
     8→    case russian = "ru"
     9→    case english = "en"
    10→    case chinese = "zh"
    11→
    12→    var id: String { rawValue }
    13→
    14→    /// Название языка на самом языке (для переключателя).
    15→    var nativeName: String {
    16→        switch self {
    17→        case .russian: return "Русский"
    18→        case .english: return "English"
    19→        case .chinese: return "中文"
    20→        }
    21→    }
    22→
    23→    /// Флаг для иконки.
    24→    var flag: String {
    25→        switch self {
    26→        case .russian: return "🇷🇺"
    27→        case .english: return "🇬🇧"
    28→        case .chinese: return "🇨🇳"
    29→        }
    30→    }
    31→}

    33→// MARK: - Localization Manager
    34→/// Менеджер локализации с переключением языка в рантайме.
    35→/// Хранит выбор в UserDefaults, транслирует изменения через objectWillChange.
    36→@MainActor
    37→final class LocalizationManager: ObservableObject {
    38→
    39→    static let shared = LocalizationManager()
    40→
    41→    /// Текущий язык (публикуется → UI обновляется автоматически).
    42→    @Published var currentLanguage: AppLanguage {
    43→        didSet {
    44→            UserDefaults.standard.set(currentLanguage.rawValue, forKey: Self.storageKey)
    45→        }
    46→    }
    47→
    48→    private static let storageKey = "plink_app_language"
    49→
    50→    private init() {
    51→        let raw = UserDefaults.standard.string(forKey: Self.storageKey) ?? AppLanguage.russian.rawValue
    52→        currentLanguage = AppLanguage(rawValue: raw) ?? .russian
    53→    }
    54→
    55→    /// Nonisolated доступ к текущему языку для use-case'ов вне MainActor
    56→    /// (например, вычисляемые свойства enum'ов). Читает напрямую из UserDefaults.
    57→    static var sharedSafe: LanguageReader { LanguageReader() }
    58→
    59→    /// Локализованная строка по ключу.
    60→    func string(_ key: L10n.Key) -> String {
    61→        L10n.table[key]?[currentLanguage] ?? key.rawValue
    62→    }
    63→}
    64→
    65→/// Thread-safe read-only доступ к выбранному языку.
    66→struct LanguageReader {
    67→    var currentLanguage: AppLanguage {
    68→        let raw = UserDefaults.standard.string(forKey: "plink_app_language") ?? AppLanguage.russian.rawValue
    69→        return AppLanguage(rawValue: raw) ?? .russian
    70→    }
    71→}
