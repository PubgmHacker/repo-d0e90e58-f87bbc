    10‚Üí    10‚Üí/// ‚Ēā                   THE LATENCY PROBLEM                            ‚Ēā
    11‚Üí    11‚Üí/// ‚Ēā                                                                  ‚Ēā
    12‚Üí    12‚Üí/// ‚Ēā  Host presses Play at media-time = 10.0s, wall-clock T0.        ‚Ēā
    13‚Üí    13‚Üí/// ‚Ēā  Command travels to server (T0 + uplink)                         ‚Ēā
    14‚Üí    14‚Üí/// ‚Ēā  Server broadcasts to all clients (T0 + uplink + downlink)       ‚Ēā
    15‚Üí    15‚Üí/// ‚Ēā  Client receives command at T0 + RTT                              ‚Ēā
    16‚Üí    16‚Üí/// ‚Ēā                                                                  ‚Ēā
    17‚Üí    17‚Üí/// ‚Ēā  If client seeks to 10.0s and plays, it is already RTT seconds   ‚Ēā
    18‚Üí    18‚Üí/// ‚Ēā  behind the host. Everyone sees different frames.                ‚Ēā
    19‚Üí    19‚Üí///                                                                  ‚Ēā
    20‚Üí    20‚Üí/// ‚Ēā  SOLUTION: compensate using synchronized server clock + RTT.     ‚Ēā
    21‚Üí    21‚Üí/// ‚ĒĒ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚ĒÄ‚Ēė
    22‚Üí    22‚Üí///

    24‚Üí    24‚Üí///   elapsedSinceEvent = currentServerTime - eventServerTimestamp

   203‚Üí   203‚Üí        guard isHost else { return }

   215‚Üí   215‚Üí        player?.seek(to: CMTime(seconds: clamped, preferredTimescale: 600),
   216‚Üí   216‚Üí                     completionHandler: handler)
   217‚Üí   217‚Üí
   218‚Üí   218‚Üí        // ūüĒß FIX 2.3: Fallback timeout ‚ÄĒ if seek doesn't complete in 2s, broadcast anyway

   225‚Üí   225‚Üí            self.seekCompletionHandler = nil
   226‚Üí   226‚Üí        }
   227‚Üí   227‚Üí    }

   244‚Üí   244‚Üí
   245‚Üí   245‚Üí    func handleSyncMessage(_ message: SyncMessage) {
   246‚Üí   246‚Üí        // Ignore echoes of our own commands (server broadcasts to all incl. sender)
   247‚Üí   247‚Üí        guard message.senderID != userID else { return }
   248‚Üí   248‚Üí
   249‚Üí   249‚Üí        switch message.command {
   250‚Üí   250‚Üí        case .play:
   251‚Üí   251‚Üí            handlePlay(message)
   252‚Üí   252‚Üí        case .pause:
   253‚Üí   253‚Üí            handlePause(message)
   254‚Üí   254‚Üí        case .seek:
   255‚Üí   255‚Üí            handleSeek(message)
   256‚Üí   256‚Üí        case .changeMedia:
   257‚Üí   257‚Üí            if let item = message.mediaItem {
   258‚Üí   258‚Üí                handleMediaChange(item, from: message)
   259‚Üí   259‚Üí            }
   260‚Üí   260‚Üí        case .stateRequest:
   261‚Üí   261‚Üí            // A participant is asking the host for the current state
   262‚Üí   262‚Üí            if isHost {
   263‚Üí   263‚Üí                respondWithCurrentState()
   264‚Üí   264‚Üí            }
   265‚Üí   265‚Üí        case .stateResponse:
   266‚Üí   266‚Üí            handleStateResponse(message)
   267‚Üí   267‚Üí        case .correction:
   268‚Üí   268‚Üí            handleForcedCorrection(message)
   269‚Üí   269‚Üí        case .ping, .pong:
   270‚Üí   270‚Üí            break
   271‚Üí   271‚Üí        }
   272‚Üí   272‚Üí    }

   295‚Üí   295‚Üí    // MARK: - Host: Periodic State Broadcast
   296‚Üí   296‚Üí    /// Every 2s the host sends its current playback position stamped with
   297‚Üí   297‚Üí    /// server time. Participants use this to detect and correct drift.
   298‚Üí   298‚Üí    private func broadcastPeriodicState() {
   299‚Üí   299‚Üí        guard isHost else { return }
   300‚Üí   300‚Üí
   301‚Üí   301‚Üí        // Use a lightweight "seek" envelope as a heartbeat state sync.
   302‚Üí   302‚Üí        // Participants ignore seek commands within tolerance, but reseek
   303‚Üí   303‚Üí        // if drift exceeds the threshold.
   304‚Üí   304‚Üí        let msg = SyncMessage(
   305‚Üí   305‚Üí            command: .seek,        // reuse seek as a state pulse

   333‚Üí   333‚Üí    /// Compare local playback position against the extrapolated host position.
   334‚Üí   334‚Üí    /// If drift exceeds threshold, self-correct (participant) or request state (host).
   335‚Üí   335‚Üí    private func checkDrift() {
   336‚Üí   336‚Üí        guard lastSyncEventTime > 0 else { return }
   337‚Üí   337‚Üí
   338‚Üí   338‚Üí        // Extrapolate where the host SHOULD be right now
   339‚Üí   339‚Üí        let elapsed = currentServerTime - lastSyncEventTime
   340‚Üí   340‚Üí        let extrapolatedHostTime: TimeInterval
   341‚Üí   341‚Üí        if lastSyncWasPlaying {
   342‚Üí   342‚Üí            extrapolatedHostTime = min(lastSyncMediaTime + elapsed, duration)
   343‚Üí   343‚Üí        } else {
   344‚Üí   344‚Üí            extrapolatedHostTime = lastSyncMediaTime
   345‚Üí   345‚Üí        }
   346‚Üí   346‚Üí
   347‚Üí   347‚Üí        let drift = abs(currentTime - extrapolatedHostTime)
   348‚Üí   348‚Üí        let driftMs = drift * 1000
   349‚Üí   349‚Üí        syncQuality = .fromDrift(driftMs)
   350‚Üí   350‚Üí
   351‚Üí   351‚Üí        // Surface telemetry
   352‚Üí   352‚Üí        estimatedRTTms = Int(estimatedRTT * 1000)
   353‚Üí   353‚Üí
   354‚Üí   354‚Üí        // Hard resync: participant drifted way off ‚ÄĒ request fresh state from host
   355‚Üí   355‚Üí        if !isHost && drift > Constants.hardResyncThreshold {
   356‚Üí   356‚Üí            Logger.sync.warn("Hard drift: \(String(format: "%.0f", driftMs))ms ‚ÄĒ requesting state from host")
   357‚Üí   357‚Üí            requestStateFromHost()
   358‚Üí   358‚Üí            return
   359‚Üí   359‚Üí        }
   360‚Üí   360‚Üí
   361‚Üí   361‚Üí        // Soft correction: small drift, nudge locally without a visible jump
   362‚Üí   362‚Üí        if !isHost && drift > Constants.driftThreshold && drift <= Constants.hardResyncThreshold {
   363‚Üí   363‚Üí            Logger.sync.info("Soft drift: \(String(format: "%.0f", driftMs))ms ‚ÄĒ self-correcting to \(String(format: "%.2f", extrapolatedHostTime))s")
   364‚Üí   364‚Üí            seekSilently(to: extrapolatedHostTime, preserveRate: isPlaying)
   365‚Üí   365‚Üí            lastCompensationMs = Int(driftMs)
   366‚Üí   366‚Üí        }
   367‚Üí   367‚Üí    }

   369‚Üí   369‚Üí    // MARK: - Latency-Compensated Command Handlers
   370‚Üí   370‚Üí
   371‚Üí   371‚Üí    /// PLAY command received.
   372‚Üí   372‚Üí    /// The host stamped this with (mediaTime, serverTimestamp).
   373‚Üí   373‚Üí    /// By the time we receive it, real playback on the host has advanced by
   374‚Üí   374‚Üí    /// (currentServerTime - serverTimestamp). We must seek AHEAD by that delta
   375‚Üí   375‚Üí    /// so we're aligned with where the host actually is right now.
   376‚Üí   376‚Üí    private func handlePlay(_ message: SyncMessage) {
   377‚Üí   377‚Üí        let eventMediaTime = message.mediaTime ?? 0
   378‚Üí   378‚Üí        let eventServerTime = message.timestamp
   379‚Üí   379‚Üí
   380‚Üí   380‚Üí        // ‚ĒÄ‚ĒÄ‚ĒÄ Core latency compensation ‚ĒÄ‚ĒÄ‚ĒÄ
   381‚Üí   381‚Üí        let elapsedSinceEvent = max(0, currentServerTime - eventServerTime)
   382‚Üí   382‚Üí        let compensatedTarget = min(
   383‚Üí   383‚Üí            eventMediaTime + min(elapsedSinceEvent, Constants.maxPredictiveJump),
   384‚Üí   384‚Üí            duration > 0 ? duration : .infinity
   385‚Üí   385‚Üí        )
   386‚Üí   386‚Üí
   387‚Üí   387‚Üí        let compensationMs = Int(elapsedSinceEvent * 1000)
   388‚Üí   388‚Üí        lastCompensationMs = compensationMs
   389‚Üí   389‚Üí        recordSyncPoint(mediaTime: compensatedTarget, isPlaying: true, serverTime: currentServerTime)
   390‚Üí   390‚Üí
   391‚Üí   391‚Üí        Logger.sync.info("‚Ė∂ÔłŹ PLAY  host@\(fmt(eventMediaTime))s ‚Üí seek to \(fmt(compensatedTarget))s (+\(compensationMs)ms latency, RTT \(Int(estimatedRTT*1000))ms)")
   392‚Üí   392‚Üí
   393‚Üí   393‚Üí        // ūüĒß FAST PATH: If we're already playing and within tolerance, do nothing
   394‚Üí   394‚Üí        // (avoids stutter on rapid play/pause toggles)
   395‚Üí   395‚Üí        if isPlaying && abs(currentTime - compensatedTarget) < Constants.seekTolerance {
   396‚Üí   396‚Üí            return
   397‚Üí   397‚Üí        }
   398‚Üí   398‚Üí
   399‚Üí   399‚Üí        // ūüĒß FAST PATH: If drift is small (< 2s), just play without seeking ‚ÄĒ
   400‚Üí   400‚Üí        // the drift monitor will self-correct within a few seconds.
   401‚Üí   401‚Üí        // This avoids the "seek then play" stutter that was causing delay.
   402‚Üí   402‚Üí        if abs(currentTime - compensatedTarget) < 2.0 {
   403‚Üí   403‚Üí            player?.play()
   404‚Üí   404‚Üí            isPlaying = true
   405‚Üí   405‚Üí            return
   406‚Üí   406‚Üí        }
   407‚Üí   407‚Üí
   408‚Üí   408‚Üí        // Large drift ‚ÄĒ seek to compensated position, then play
   409‚Üí   409‚Üí        player?.seek(to: CMTime(seconds: compensatedTarget, preferredTimescale: 600)) { [weak self] _ in
   410‚Üí   410‚Üí            guard let self else { return }
   411‚Üí   411‚Üí            Task { @MainActor in
   412‚Üí   412‚Üí                self.player?.play()
   413‚Üí   413‚Üí                self.isPlaying = true
   414‚Üí   414‚Üí                self.currentTime = compensatedTarget
   415‚Üí   415‚Üí            }
   416‚Üí   416‚Üí        }
   417‚Üí   417‚Üí    }
   418‚Üí   418‚Üí
   419‚Üí   419‚Üí    /// PAUSE command received.
   420‚Üí   420‚Üí    /// ūüĒß IMMEDIATE: Pause instantly for zero-latency visual sync.
   421‚Üí   421‚Üí    /// Seek to exact frame happens after pause (non-blocking).

   439‚Üí   439‚Üí    /// SEEK command received (also used as periodic state pulse from host).
   440‚Üí   440‚Üí    /// Compensate for latency so we land where the host actually is now.
   441‚Üí   441‚Üí    ///
   442‚Üí   442‚Üí    /// ūüĒß FIX M1: Was using `elapsedSinceEvent < stateBroadcastInterval + 1` as a
   443‚Üí   443‚Üí    /// heuristic to detect state pulses vs real seeks. But the host's periodic
   444‚Üí   444‚Üí    /// state broadcast (every 2s) and a user-initiated seek can both arrive within
   445‚Üí   445‚Üí    /// that 3s window, causing real seeks to be silently reverted (extrapolated
   446‚Üí   446‚Üí    /// back to the original position).
   447‚Üí   447‚Üí    ///
   448‚Üí   448‚Üí    /// New heuristic: a state pulse is a seek where |eventMediaTime - lastSyncMediaTime|
   449‚Üí   449‚Üí    /// is small (‚Č§ 2s, normal playback progress). A real seek has a large jump.
   450‚Üí   450‚Üí    /// This distinguishes "host is just broadcasting position" from "host actually
   451‚Üí   451‚Üí    /// jumped to a new position".
   452‚Üí   452‚Üí    private func handleSeek(_ message: SyncMessage) {
   453‚Üí   453‚Üí        guard let eventMediaTime = message.mediaTime else { return }
   454‚Üí   454‚Üí        let eventServerTime = message.timestamp
   455‚Üí   455‚Üí        let elapsedSinceEvent = max(0, currentServerTime - eventServerTime)
   456‚Üí   456‚Üí
   457‚Üí   457‚Üí        // ūüĒß FIX M1: Detect state pulse by comparing media time delta, NOT time elapsed.
   458‚Üí   458‚Üí        // A state pulse has media time ‚Čą lastSyncMediaTime + (elapsed * playbackRate).
   459‚Üí   459‚Üí        // A real seek has a large jump in media time relative to lastSyncMediaTime.
   460‚Üí   460‚Üí        let expectedProgress = isPlaying ? elapsedSinceEvent : 0
   461‚Üí   461‚Üí        let mediaTimeDelta = abs(eventMediaTime - (lastSyncMediaTime + expectedProgress))
   462‚Üí   462‚Üí        let isStatePulse = mediaTimeDelta < Constants.stateBroadcastInterval  // ‚Č§ 2s drift = pulse
   463‚Üí   463‚Üí
   464‚Üí   464‚Üí        let compensatedTarget: TimeInterval
   465‚Üí   465‚Üí        if isStatePulse && isPlaying {
   466‚Üí   466‚Üí            compensatedTarget = min(eventMediaTime + elapsedSinceEvent, duration > 0 ? duration : .infinity)
   467‚Üí   467‚Üí        } else {
   468‚Üí   468‚Üí            // Real seek ‚ÄĒ don't extrapolate, just go to the exact position.
   469‚Üí   469‚Üí            compensatedTarget = eventMediaTime
   470‚Üí   470‚Üí        }
   471‚Üí   471‚Üí
   472‚Üí   472‚Üí        // Within tolerance ‚Üí ignore (prevents jitter from periodic pulses)
   473‚Üí   473‚Üí        if abs(currentTime - compensatedTarget) < Constants.seekTolerance {
   474‚Üí   474‚Üí            recordSyncPoint(mediaTime: compensatedTarget, isPlaying: isPlaying, serverTime: currentServerTime)
   475‚Üí   475‚Üí            return
   476‚Üí   476‚Üí        }
   477‚Üí   477‚Üí
   478‚Üí   478‚Üí        recordSyncPoint(mediaTime: compensatedTarget, isPlaying: isPlaying, serverTime: currentServerTime)
   479‚Üí   479‚Üí        Logger.sync.info("‚Ź© SEEK  host@\(fmt(eventMediaTime))s ‚Üí \(fmt(compensatedTarget))s [\(isStatePulse ? "pulse" : "real-seek")]")
   480‚Üí   480‚Üí
   481‚Üí   481‚Üí        let wasPlaying = isPlaying
   482‚Üí   482‚Üí        seekSilently(to: compensatedTarget, preserveRate: wasPlaying)
   483‚Üí   483‚Üí    }

   616‚Üí   616‚Üí        if let observer = bufferObserver {
   617‚Üí   617‚Üí            NotificationCenter.default.removeObserver(observer)

   619‚Üí   619‚Üí        }
   620‚Üí   620‚Üí        player?.pause()

   622‚Üí   622‚Üí        player = nil
   623‚Üí   623‚Üí        playerItem = nil
   624‚Üí   624‚Üí        currentMediaItem = nil
   625‚Üí   625‚Üí        isPlaying = false
   626‚Üí   626‚Üí        currentTime = 0
   627‚Üí   627‚Üí        duration = 0
   628‚Üí   628‚Üí        stopDriftMonitor()
   629‚Üí   629‚Üí        stopStateBroadcast()
   630‚Üí   630‚Üí    }

   660‚Üí   660‚Üí    // ūüĒß FIX 1.2: Buffer underrun observer ‚ÄĒ detects when AVPlayer stalls
   661‚Üí   661‚Üí    // due to slow network. Pauses local playback (without broadcasting pause
   662‚Üí   662‚Üí    // to other participants), then resumes + requests fresh state from host
   663‚Üí   663‚Üí    // when buffer recovers. Prevents visible desync during buffering.
   664‚Üí   664‚Üí    private func observeBufferUnderrun(_ item: AVPlayerItem) {
   665‚Üí   665‚Üí        // Listen for playback stall notification
   666‚Üí   666‚Üí        bufferObserver = NotificationCenter.default.addObserver(
   667‚Üí   667‚Üí            forName: .AVPlayerItemPlaybackStalled,
   668‚Üí   668‚Üí            object: item,
   669‚Üí   669‚Üí            queue: .main
   670‚Üí   670‚Üí        ) { [weak self] _ in
   671‚Üí   671‚Üí            guard let self, !self.isHost else { return }
   672‚Üí   672‚Üí            Task { @MainActor in
   673‚Üí   673‚Üí                self.bufferUnderrunCount += 1
   674‚Üí   674‚Üí                Logger.sync.warn("Buffer underrun #\(self.bufferUnderrunCount) ‚ÄĒ local pause (no broadcast)")
   675‚Üí   675‚Üí                self.player?.pause()
   676‚Üí   676‚Üí            }
   677‚Üí   677‚Üí        }
   678‚Üí   678‚Üí
   679‚Üí   679‚Üí        // Also observe isPlaybackLikelyToKeepUp for recovery
   680‚Üí   680‚Üí        item.publisher(for: \.isPlaybackLikelyToKeepUp, options: .new)
   681‚Üí   681‚Üí            .receive(on: DispatchQueue.main)
   682‚Üí   682‚Üí            .sink { [weak self] canKeepUp in
   683‚Üí   683‚Üí                guard let self, !self.isHost else { return }
   684‚Üí   684‚Üí                if canKeepUp && self.isPlaying {
   685‚Üí   685‚Üí                    // Buffer recovered ‚ÄĒ resume local playback + request fresh state
   686‚Üí   686‚Üí                    Logger.sync.info("Buffer recovered ‚ÄĒ resuming + requesting fresh state from host")
   687‚Üí   687‚Üí                    self.player?.play()
   688‚Üí   688‚Üí                    self.requestStateFromHost()
   689‚Üí   689‚Üí                }
   690‚Üí   690‚Üí            }
   691‚Üí   691‚Üí            .store(in: &cancellables)
   692‚Üí   692‚Üí    }
