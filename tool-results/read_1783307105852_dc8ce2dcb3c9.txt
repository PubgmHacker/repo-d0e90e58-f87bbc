   543→// MARK: - Premium Animated Avatar Border
   544→struct PremiumAvatarModifier: ViewModifier {
   545→    @State private var rotation: Double = 0
   546→
   547→    func body(content: Content) -> some View {
   548→        content
   549→            .overlay(
   550→                Circle()
   551→                    .trim(from: 0, to: CGFloat(rotation / 360.0))
   552→                    .stroke(
   553→                        AngularGradient(
   554→                            colors: [
   555→                                Color.bioCyan,
   556→                                Color.raveAccent,
   557→                                Color(hex: 0x22D3EE),
   558→                                Color.bioCyan,
   559→                            ],
   560→                            center: .center
   561→                        ),
   562→                        lineWidth: 3.5
   563→                    )
   564→                    .rotationEffect(.degrees(rotation))
   565→            )
   566→            .onAppear {
   567→                withAnimation(.linear(duration: 4).repeatForever(autoreverses: false)) {
   568→                    rotation = 360
   569→                }
   570→            }
   571→    }
   572→}
