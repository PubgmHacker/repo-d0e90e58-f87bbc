   567→    /// 🔧 Рекомендации: на основе ИИ + история просмотров + разные сервисы
   568→    private var recommendationRooms: [Room] {
   569→        // Сортируем по активности, берём разные сервисы
   570→        let all = viewModel.filteredRooms.sorted { $0.participantCount > $1.participantCount }
   571→        // Группируем по сервису чтобы получить разнообразие
   572→        var seen = Set<VideoService>()
   573→        var result: [Room] = []
   574→        for room in all {
   575→            if let source = room.mediaItem?.source, let service = VideoService(rawValue: source.rawValue), !seen.contains(service) {
   576→                seen.insert(service)
   577→                result.append(room)
   578→            }
   579→        }
   580→        // Добавляем остальные
   581→        result.append(contentsOf: all.filter { room in
   582→            !result.contains(where: { $0.id == room.id })
   583→        })
   584→        return Array(result.prefix(10))
   585→    }

   658→    private var isValidJoinInput: Bool {
   659→        let trimmed = joinInput.trimmingCharacters(in: .whitespaces)
   660→        return trimmed.count == 6 || trimmed.contains("http") || trimmed.contains("plnk") || trimmed.contains(".")
   661→    }
   662→
   663→    private func extractCode(from input: String) -> String {
   664→        let trimmed = input.trimmingCharacters(in: .whitespaces)
   665→        if trimmed.count == 6 && trimmed.allSatisfy({ $0.isLetter || $0.isNumber }) { return trimmed }
   666→        if let url = URL(string: trimmed) {
   667→            if let code = URLComponents(url: url, resolvingAgainstBaseURL: false)?.queryItems?.first(where: { $0.name == "code" })?.value { return code }
   668→            let path = url.lastPathComponent
   669→            if path.count >= 4 && path.count <= 8 { return path.uppercased() }
   670→        }
   671→        return trimmed
   672→    }
