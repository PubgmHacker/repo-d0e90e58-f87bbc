   758→    private var mediaIcon: String {
   759→        switch item.mediaType {

   762→        case "music": return "music.note"

   768→    @ViewBuilder
   769→    private var posterImage: some View {
   770→        if let urlStr = item.thumbnailURL, let url = URL(string: urlStr) {
   771→            AsyncImage(url: url) { phase in
   772→                if let image = phase.image {
   773→                    image.resizable().aspectRatio(contentMode: .fill)
   774→                } else {
   775→                    posterGradient
   776→                }
   777→            }
   778→        } else {
   779→            posterGradient
   780→        }
   781→    }
   782→
   783→    private var posterGradient: some View {
   784→        LinearGradient(
   785→            colors: gradientPalette,
   786→            startPoint: .topLeading,
   787→            endPoint: .bottomTrailing
   788→        )
   789→        .overlay(
   790→            Image(systemName: "play.circle.fill")
   791→                .font(.system(size: 28))
   792→                .foregroundColor(.white.opacity(0.6))
   793→        )
   794→    }
   795→
   796→    private var gradientPalette: [Color] {
   797→        let palettes: [[Color]] = [
   798→            [Color.ravePrimary.opacity(0.4), .black],
   799→            [Color.raveAccent.opacity(0.4), .black],
   800→            [Color.raveCyan.opacity(0.4), .black],
   801→            [Color.raveWarning.opacity(0.4), .black],
   802→        ]
   803→        return palettes[abs(item.id.hashValue) % palettes.count]
   804→    }
