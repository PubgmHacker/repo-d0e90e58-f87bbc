   236→// MARK: - VideoService + Auth Requirements
   237→
   238→extension VideoService {
   239→    /// 🔧 Returns true if this service requires authentication/subscription for content.
   240→    /// YouTube, VK Video, and Rutube have free public content.
   241→    /// Cinema services require subscription.
   242→    var requiresAuth: Bool {
   243→        switch self {
   244→        case .youtube, .vk, .rutube, .smotrim, .browser, .customURL:
   245→            return false  // free / public content
   246→        case .netflix, .disney, .kinopoisk, .ivi, .okko, .wink, .start, .premier, .kion:
   247→            return true   // subscription required
   248→        }
   249→    }
   250→
   251→    /// 🔧 Detects if a URL is a video page for this service, and returns the
   252→    /// embeddable video URL if so. Returns nil if the URL is not a video page.
   253→    static func detectVideoURL(_ url: URL, for service: VideoService, title: String?) -> DetectedVideo? {
   254→        let urlString = url.absoluteString
   255→        let host = url.host ?? ""
   256→
   257→        switch service {
   258→        case .youtube:
   259→            // youtube.com/watch?v=VIDEO_ID or youtu.be/VIDEO_ID
   260→            if host.contains("youtube.com") {
   261→                if let components = URLComponents(url: url, resolvingAgainstBaseURL: false),
   262→                   let videoId = components.queryItems?.first(where: { $0.name == "v" })?.value {
   263→                    return DetectedVideo(
   264→                        title: title,
   265→                        embedURL: "https://www.youtube.com/embed/\(videoId)",
   266→                        originalURL: urlString,
   267→                        service: .youtube
   268→                    )
   269→                }
   270→            }
   271→            if host.contains("youtu.be") {
   272→                let videoId = url.lastPathComponent
   273→                if !videoId.isEmpty {
   274→                    return DetectedVideo(
   275→                        title: title,
   276→                        embedURL: "https://www.youtube.com/embed/\(videoId)",
   277→                        originalURL: urlString,
   278→                        service: .youtube
   279→                    )
   280→                }
   281→            }
   282→
   283→        case .vk:
   284→            // vk.com/video-OWNER_ID_VIDEO_ID or vk.com/video/OWNER_ID_VIDEO_ID
   285→            if host.contains("vk.com") || host.contains("vk.ru") {
   286→                let path = url.path
   287→                if path.contains("/video") {
   288→                    // For VK, we use the original URL — VK's player works in WebView
   289→                    return DetectedVideo(
   290→                        title: title,
   291→                        embedURL: urlString,
   292→                        originalURL: urlString,
   293→                        service: .vk
   294→                    )
   295→                }
   296→            }
   297→
   298→        case .rutube:
   299→            // rutube.ru/video/VIDEO_ID/
   300→            if host.contains("rutube.ru") || host.contains("rutube.video") {
   301→                let path = url.path
   302→                if path.contains("/video/") {
   303→                    // Extract video ID from path
   304→                    let segments = path.split(separator: "/")
   305→                    if segments.count >= 2, segments[0] == "video" {
   306→                        let videoId = String(segments[1])
   307→                        return DetectedVideo(
   308→                            title: title,
   309→                            embedURL: "https://rutube.ru/play/embed/\(videoId)",
   310→                            originalURL: urlString,
   311→                            service: .rutube
   312→                        )
   313→                    }
   314→                }
   315→            }
   316→
   317→        case .kinopoisk, .ivi, .okko, .wink, .start, .premier, .smotrim, .kion, .netflix, .disney:
   318→            // For cinema services, detect video/content pages by URL patterns
   319→            // These services use their own players with DRM — we can't extract streams
   320→            // Instead, the page URL itself becomes the content URL
   321→            // The WebView will be the player in the room
   322→            let path = url.path.lowercased()
   323→            // Common patterns: /film/, /series/, /video/, /watch/, /play/
   324→            let videoPatterns = ["/film/", "/series/", "/video/", "/watch/", "/play/", "/movies/", "/show/"]
   325→            if videoPatterns.contains(where: { path.contains($0) }) {
   326→                return DetectedVideo(
   327→                    title: title,
   328→                    embedURL: urlString,  // use original URL — WebView is the player
   329→                    originalURL: urlString,
   330→                    service: service
   331→                )
   332→            }
   333→
   334→        case .browser, .customURL:
   335→            // No auto-detection for browser/customURL
   336→            break
   337→        }
   338→
   339→        return nil
   340→    }
   341→}

   456→    final class Coordinator: NSObject, WKNavigationDelegate, WKUIDelegate, WKScriptMessageHandler {
   457→        let parent: ServiceWebView
   458→        private var lastDetectedURL: String?
   459→
   460→        init(parent: ServiceWebView) {

   464→        // 🔧 Pack v3: Handle SPA URL changes from JavaScript
   465→        func userContentController(_ userContentController: WKUserContentController, didReceive message: WKScriptMessage) {
   466→            if message.name == "plinkURLChange",
   467→               let body = message.body as? [String: Any],
   468→               let urlString = body["url"] as? String,
   469→               let url = URL(string: urlString) {
   470→
   471→                let title = body["title"] as? String
   472→                let service = Self.serviceFromURL(url)
   473→                if let service, let detected = VideoService.detectVideoURL(url, for: service, title: title) {
   474→                    DispatchQueue.main.async {
   475→                        self.parent.currentURL = url
   476→                        self.parent.pageTitle = title ?? ""
   477→                        self.parent.onVideoDetected?(detected)
   478→                    }
   479→                }
   480→            }
   481→        }
   482→
   483→        // 🔧 Pack v3: Перехватываем КАЖДУЮ навигацию (включая SPA YouTube).
   484→        // Раньше: только didFinish → YouTube SPA не триггерит → видео играло без создания комнаты.
   485→        // Теперь: decidePolicyFor ловит URLchange → детектим видео → авто-переход.
   486→        func webView(_ webView: WKWebView, decidePolicyFor navigationAction: WKNavigationAction, decisionHandler: @escaping (WKNavigationActionPolicy) -> Void) {
   487→            // Сначала проверяем URL на видео
   488→            if let url = navigationAction.request.url {
   489→                let urlString = url.absoluteString
   490→                // Избегаем повторного срабатывания на тот же URL
   491→                if urlString != lastDetectedURL {
   492→                    lastDetectedURL = urlString
   493→                    let service = Self.serviceFromURL(url)
   494→                    if let service, let detected = VideoService.detectVideoURL(url, for: service, title: nil) {
   495→                        // 🔧 FIX: CANCEL navigation — don't load the YouTube watch page.
   496→                        // Was: .allow → YouTube watch page loaded → video auto-played for
   497→                        // a second before room creation took over.
   498→                        // Now: .cancel → YouTube watch page never loads, no video playback.
   499→                        // We just extract the video ID from the URL and create the room.
   500→                        DispatchQueue.main.async {
   501→                            self.parent.currentURL = url
   502→                            self.parent.pageTitle = webView.title ?? ""
   503→                            self.parent.onVideoDetected?(detected)
   504→                        }
   505→                        decisionHandler(.cancel)
   506→                        return
   507→                    }
   508→                }
   509→            }
   510→            decisionHandler(.allow)
   511→        }
   512→
   513→        func webView(_ webView: WKWebView, didFinish navigation: WKNavigation!) {
   514→            DispatchQueue.main.async {
   515→                self.parent.currentURL = webView.url
   516→                self.parent.pageTitle = webView.title ?? ""
   517→                self.parent.canGoBack = webView.canGoBack
   518→                self.parent.canGoForward = webView.canGoForward
   519→            }
   520→
   521→            // 🔧 Detect video page on full load too
   522→            if let url = webView.url {
   523→                let title = webView.title
   524→                let service = Self.serviceFromURL(url)
   525→                if let service, let detected = VideoService.detectVideoURL(url, for: service, title: title) {
   526→                    DispatchQueue.main.async {
   527→                        self.parent.onVideoDetected?(detected)
   528→                    }
   529→                }
   530→            }
   531→
   532→            // 🔧 Pack v3: Inject JS to detect SPA URL changes (YouTube React app)
   533→            // YouTube меняет URL через History API без полной перезагрузки.
   534→            // Этот скрипт следит за URL и вызывает Swift когда URL меняется.
   535→            let js = """
   536→            (function() {
   537→                if (window._plinkURLObserver) return;
   538→                window._plinkURLObserver = true;
   539→                let lastURL = window.location.href;
   540→                setInterval(function() {
   541→                    if (window.location.href !== lastURL) {
   542→                        lastURL = window.location.href;

   576→            // 🔧 FIX: Rutube opens video in new window. Check for video URL before loading.
   577→            if let url = navigationAction.request.url {

   580→                    DispatchQueue.main.async {

   593→        // 🔧 Helper: determine VideoService from URL host
   594→        private static func serviceFromURL(_ url: URL) -> VideoService? {
   595→            let host = (url.host ?? "").lowercased()
   596→            if host.contains("youtube") || host.contains("youtu.be") { return .youtube }
   597→            if host.contains("vk.com") || host.contains("vk.ru") { return .vk }
   598→            if host.contains("rutube") { return .rutube }
   599→            if host.contains("netflix") { return .netflix }
   600→            if host.contains("disney") { return .disney }
   601→            if host.contains("kinopoisk") { return .kinopoisk }
   602→            if host.contains("ivi.ru") || host.contains("ivi.tv") { return .ivi }
   603→            if host.contains("okko") { return .okko }
   604→            if host.contains("wink") { return .wink }
   605→            if host.contains("start.ru") { return .start }
   606→            if host.contains("premier") { return .premier }
   607→            if host.contains("smotrim") { return .smotrim }
   608→            if host.contains("kion") { return .kion }
   609→            return nil
   610→        }
