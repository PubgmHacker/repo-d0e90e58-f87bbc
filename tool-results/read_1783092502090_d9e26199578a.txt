   510→    /// 🔧 Live-комнаты: топ 5-10 самых активных по онлайну, отсортированных по participantCount
   511→    private var liveRooms: [Room] {
   512→        viewModel.filteredRooms
   513→            .filter { $0.isActive && $0.privacy == .publicRoom }
   514→            .sorted { $0.participantCount > $1.participantCount }
   515→            .prefix(10)
   516→            .map { $0 }

   519→    /// 🔧 Рекомендации: на основе ИИ + история просмотров + разные сервисы
   520→    private var recommendationRooms: [Room] {
   521→        // Сортируем по активности, берём разные сервисы
   522→        let all = viewModel.filteredRooms.sorted { $0.participantCount > $1.participantCount }
   523→        // Группируем по сервису чтобы получить разнообразие
   524→        var seen = Set<VideoService>()
   525→        var result: [Room] = []
   526→        for room in all {
   527→            if let service = room.mediaItem?.source, !seen.contains(service) {
   528→                seen.insert(service)
   529→                result.append(room)
   530→            }
   531→        }
   532→        // Добавляем остальные
   533→        result.append(contentsOf: all.filter { room in
   534→            !result.contains(where: { $0.id == room.id })
   535→        })
   536→        return Array(result.prefix(10))
   537→    }

   583→                Text(loc.string(.joinEnter))
   584→                    .frame(maxWidth: .infinity)
   585→            }
   586→            .buttonStyle(PremiumButtonStyle())
   587→            .background(Color.raveGradient)
   588→            .clipShape(RoundedRectangle(cornerRadius: 16))
   589→            .disabled(!isValidJoinInput || viewModel.isLoading)
   590→            .padding(.horizontal, 40)
   591→
   592→            if let error = viewModel.errorMessage {
   593→                Text(error).font(.caption).foregroundColor(.raveDanger)
   594→            }
   595→
   596→            Button(loc.string(.cancel)) {
   597→                showJoinSheet = false
   598→                joinInput = ""
   599→            }
   600→            .foregroundColor(.raveTextSecondary)
   601→
   602→            Spacer()
   603→        }
   604→        .padding(.top, 32)
   605→        .presentationDetents([.medium])
   606→        .presentationBackground(Color.raveBackground)
   607→        .preferredColorScheme(.dark)
   608→    }
   609→
   610→    private var isValidJoinInput: Bool {
   611→        let trimmed = joinInput.trimmingCharacters(in: .whitespaces)
   612→        return trimmed.count == 6 || trimmed.contains("http") || trimmed.contains("plnk") || trimmed.contains(".")
   613→    }
   614→
   615→    private func extractCode(from input: String) -> String {
   616→        let trimmed = input.trimmingCharacters(in: .whitespaces)
   617→        if trimmed.count == 6 && trimmed.allSatisfy({ $0.isLetter || $0.isNumber }) { return trimmed }
   618→        if let url = URL(string: trimmed) {
   619→            if let code = URLComponents(url: url, resolvingAgainstBaseURL: false)?.queryItems?.first(where: { $0.name == "code" })?.value { return code }
   620→            let path = url.lastPathComponent
   621→            if path.count >= 4 && path.count <= 8 { return path.uppercased() }
   622→        }
   623→        return trimmed
   624→    }
